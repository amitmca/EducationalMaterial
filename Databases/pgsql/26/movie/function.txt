	CREATE OR REPLACE FUNCTION ACTOR_MOVIE() RETURNS BOOLEAN AS '
	DECLARE
	CUR_NAME CURSOR FOR
	SELECT MNAME,ANAME,RATE FROM ACTOR AS A,MOVIE AS M,MA AS R WHERE M.MNO=R.MNO AND A.ANO=R.ANO 		AND RATE>2000000; 
	ACT_NAME ACTOR.ANAME%TYPE;  
	MOV_NAME MOVIE.MNAME%TYPE; 
	BEGIN
	OPEN CUR_NAME;
	LOOP
	FETCH CUR_NAME INTO ACT_NAME,MOV_NAME;
	EXIT WHEN NOT FOUND;
	RAISE NOTICE ''%   %'',ACT_NAME,MOV_NAME;
	END LOOP;
	CLOSE CUR_NAME;
	RETURN TRUE;
	END;
	'LANGUAGE 'plpgsql';
